REM COMMENT: Switch Disk 1 into its ONLINE state
SELECT DISK 1
ONLINE DISK NOERR

REM COMMENT: Switch Disk 2 into its ONLINE state
SELECT DISK 2
ONLINE DISK NOERR

REM LABEL: VHD Pool Segment
REM COMMENT: Select VHD Pool Segment - Disk 1
SELECT VDISK FILE="H:\210 GB Part.vhdx"
ATTACH VDISK NOERR

REM LABEL: VHD Pool Segment
REM COMMENT: Select Second VHD Pool Segment - Disk 2
SELECT VDISK FILE="I:\Pool.vhdx"
ATTACH VDISK NOERR

REM LABEL: Virtual Hard Disk
SELECT VDISK FILE="H:\121G VIRTUAL.vhd"
ATTACH VDISK NOERR
ASSIGN LETTER=F NOERR

REM LABEL: Virtual Hard Disk
SELECT VDISK FILE="E:\Virtual Storage.vhdx"
ATTACH VDISK NOERR
ASSIGN LETTER=U NOERR
